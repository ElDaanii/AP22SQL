function addTarea(){
    if ($_SERVER["REQUEST_METHOD"] == "POST") {        

        $title = $_POST["title"];
        $description = $_POST["description"];
        $dueDate = $_POST["dueDate"];

        setlocale(LC_TIME, 'es_ES.UTF-8');
        $fecha_actual = date("Y-m-d");

        $datos = [
            $title,
            $description,
            $dueDate,
            $fecha_actual
        ];

        $file = fopen("tareas.csv", "a"); 
        if ($file > 0) {
            fputcsv($file, $datos); 
            fclose($file);
        } else {
            echo "Error: No se pudo abrir el archivo CSV.";
        }
    }
}
